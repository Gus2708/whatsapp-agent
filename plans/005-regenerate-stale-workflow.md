## Why this matters

`n8n_workflow.json` is the artifact the README tells operators to import, and
`setup.js` (line ~276, `n8n import:workflow --input=/etc/n8n/n8n_workflow.json`)
**imports it automatically** during `npm run setup`. But its embedded tool code is
**stale**: the `buscar_productos` and `hacer_presupuesto` nodes and the AI Agent
`systemMessage` are an older generation that lacks features present in production
(captured in `scratch_live/`): the `[PEDIR_AYUDA]` "ask an employee" flow, the
`lavaplato = fregadero` / `rieles = tubo herrería` / `fibra = macrofibra` rules,
the `×`-symbol measurement normalization, order-independent profile measures
(`40x100 == 100x40`), and the "offer in-stock substitutes when the exact item is
sold out" logic. A fresh install therefore **silently deploys a downgraded bot**.

This plan regenerates those three values from the canonical `scratch_live/` dumps
(the live source of truth) so the committed workflow matches production, and adds
a guard — run by `npm test` — that fails if they drift apart again.

## Steps

### Step 1: Create the regenerator `scripts/build_workflow.js`

```js
// Regenerate the stale code in n8n_workflow.json from the canonical scratch_live dumps.
// Only touches: buscar_productos jsCode, hacer_presupuesto jsCode, AI Agent systemMessage.
const fs = require('fs');
const path = require('path');
const root = path.join(__dirname, '..');
const read = (p) => fs.readFileSync(path.join(root, p), 'utf8').replace(/\r\n/g, '\n'); // normalize CRLF→LF (Windows checkout) for clean LF in the workflow + consistent guard

const wfPath = path.join(root, 'n8n_workflow.json');
const wf = JSON.parse(fs.readFileSync(wfPath, 'utf8'));

const buscar = read('scratch_live/live_buscar.js');
const presupuesto = read('scratch_live/live_presupuesto.js');
const systemMessage = read('scratch_live/live_systemMessage.txt');

let updated = 0;
for (const node of wf.nodes) {
  const p = node.parameters || {};
  if (p.name === 'buscar_productos') { p.jsCode = buscar; updated++; }
  else if (p.name === 'hacer_presupuesto') { p.jsCode = presupuesto; updated++; }
  else if (node.name === 'AI Agent' && p.options && 'systemMessage' in p.options) {
    p.options.systemMessage = systemMessage; updated++;
  }
}
if (updated !== 3) {
  console.error('expected to update 3 nodes, updated ' + updated + ' — aborting, no file written');
  process.exit(1);
}
fs.writeFileSync(wfPath, JSON.stringify(wf, null, 2) + '\n');
console.log('regenerated workflow: 3 nodes updated');
```

Run it.

**Verify**:
- `node scripts/build_workflow.js` → prints `regenerated workflow: 3 nodes updated`
- `node -e "JSON.parse(require('fs').readFileSync('n8n_workflow.json','utf8'));console.log('valid JSON')"` → prints `valid JSON`
- `node -e "const s=require('fs').readFileSync('n8n_workflow.json','utf8');console.log(['PEDIR_AYUDA','lavaplato','macrofibra'].every(x=>s.includes(x)))"` → prints `true`

If `build_workflow.js` prints the "expected to update 3 nodes" error, the node
names/structure don't match this plan — STOP and report (do not force it).

### Step 2: Create the guard `scripts/check_workflow_sync.js`

```js
// Guard: the code embedded in n8n_workflow.json must match the canonical scratch_live dumps.
// Fails (non-zero exit) on drift. Run via `npm test`.
const fs = require('fs');
const path = require('path');
const assert = require('assert');
const root = path.join(__dirname, '..');
const read = (p) => fs.readFileSync(path.join(root, p), 'utf8').replace(/\r\n/g, '\n'); // normalize CRLF→LF (Windows checkout) for clean LF in the workflow + consistent guard

const wf = JSON.parse(fs.readFileSync(path.join(root, 'n8n_workflow.json'), 'utf8'));
const wantJs = {
  buscar_productos: read('scratch_live/live_buscar.js'),
  hacer_presupuesto: read('scratch_live/live_presupuesto.js'),
};
const wantSys = read('scratch_live/live_systemMessage.txt');

const seen = new Set();
for (const node of wf.nodes) {
  const p = node.parameters || {};
  if (p.name && Object.prototype.hasOwnProperty.call(wantJs, p.name)) {
    assert.strictEqual(p.jsCode, wantJs[p.name],
      `n8n_workflow.json node "${p.name}" jsCode drifted from scratch_live`);
    seen.add(p.name);
  }
  if (node.name === 'AI Agent') {
    assert.strictEqual(p.options && p.options.systemMessage, wantSys,
      'AI Agent systemMessage drifted from scratch_live/live_systemMessage.txt');
    seen.add('sys');
  }
}
assert.ok(seen.has('buscar_productos') && seen.has('hacer_presupuesto') && seen.has('sys'),
  'did not find all 3 expected nodes in n8n_workflow.json');
console.log('workflow in sync with scratch_live: OK');
```

**Verify**: `node scripts/check_workflow_sync.js` → prints `workflow in sync with scratch_live: OK`, exit 0.
